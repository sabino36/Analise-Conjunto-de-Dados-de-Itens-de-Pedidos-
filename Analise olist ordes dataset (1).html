df['order_estimated_delivery_date'] = pd.to_datetime(df['order_estimated_delivery_date'])

# 1. Coluna de "Tempo de Atraso (dias)"
df['Tempo de Atraso (dias)'] = (df['order_delivered_customer_date'] - df['order_estimated_delivery_date']).dt.days

# 2. Coluna de "Tempo Total de Entrega (dias)"
df['Tempo Total de Entrega (dias)'] = (df['order_delivered_customer_date'] - df['order_purchase_timestamp']).dt.days

# 3. Coluna "Status do Pedido" (Atrasado ou No Prazo)
df['Status do Pedido'] = df['Tempo de Atraso (dias)'].apply(lambda x: 'Atrasado' if x > 0 else 'No Prazo')

# 4. Coluna "Dia da Semana da Compra" (0=Segunda-feira, 6=Domingo)
df['Dia da Semana da Compra'] = df['order_purchase_timestamp'].dt.dayofweek

# 5. Coluna "Ms da Compra"
df['Ms da Compra'] = df['order_purchase_timestamp'].dt.month

# 6. Coluna "Ano da Compra"
df['Ano da Compra'] = df['order_purchase_timestamp'].dt.year

# Agora podemos salvar a tabela com as novas colunas
df.to_csv('tabela_com_novas_colunas.csv', index=False)
